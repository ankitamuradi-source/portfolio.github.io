// ================= SMART TRAFFIC AUTOMATION SYSTEM =================
// Language: C++
// Algorithms: Dijkstra, Kruskal, BFS, Merge Sort

#include <bits/stdc++.h>
using namespace std;

#define INF INT_MAX

class Road {
public:
    int u, v, weight;
    Road(int u, int v, int w) : u(u), v(v), weight(w) {}
};

class Incident {
public:
    int location, severity;
    string description;
};

class TrafficNetwork {
public:
    int V;
    vector<vector<pair<int,int>>> adj;
    vector<Road> edges;

    TrafficNetwork(int V) {
        this->V = V;
        adj.resize(V);
    }

    void addEdge(int u, int v, int w) {
        adj[u].push_back({v,w});
        adj[v].push_back({u,w});
        edges.push_back(Road(u,v,w));
    }

    // ---------------- Dijkstra ----------------
    void dijkstra(int src) {
        vector<int> dist(V, INF);
        dist[src] = 0;
        priority_queue<pair<int,int>, vector<pair<int,int>>, greater<>> pq;
        pq.push({0, src});

        while(!pq.empty()) {
            int u = pq.top().second;
            pq.pop();
            for(auto x : adj[u]) {
                int v = x.first, w = x.second;
                if(dist[u] + w < dist[v]) {
                    dist[v] = dist[u] + w;
                    pq.push({dist[v], v});
                }
            }
        }

        cout << "\\nShortest Paths:\\n";
        for(int i=0;i<V;i++)
            cout << "Node " << i << ": " << dist[i] << endl;
    }

    // ---------------- BFS ----------------
    void bfs(int start) {
        vector<bool> visited(V,false);
        queue<int> q;
        q.push(start);
        visited[start] = true;

        cout << "\\nTraffic Flow BFS: ";
        while(!q.empty()) {
            int u = q.front(); q.pop();
            cout << u << " ";
            for(auto x: adj[u]) {
                if(!visited[x.first]) {
                    visited[x.first] = true;
                    q.push(x.first);
                }
            }
        }
        cout << endl;
    }
};

// ---------------- Merge Sort ----------------
void merge(vector<Incident> &a, int l, int m, int r) {
    int n1 = m-l+1, n2 = r-m;
    vector<Incident> L(n1), R(n2);
    for(int i=0;i<n1;i++) L[i]=a[l+i];
    for(int i=0;i<n2;i++) R[i]=a[m+1+i];

    int i=0,j=0,k=l;
    while(i<n1 && j<n2)
        a[k++] = (L[i].severity >= R[j].severity) ? L[i++] : R[j++];
    while(i<n1) a[k++]=L[i++];
    while(j<n2) a[k++]=R[j++];
}

void mergeSort(vector<Incident> &a, int l, int r) {
    if(l<r) {
        int m=(l+r)/2;
        mergeSort(a,l,m);
        mergeSort(a,m+1,r);
        merge(a,l,m,r);
    }
}

int main() {
    TrafficNetwork city(6);
    city.addEdge(0,1,4);
    city.addEdge(0,2,2);
    city.addEdge(1,3,5);
    city.addEdge(2,3,8);

    city.dijkstra(0);
    city.bfs(0);

    vector<Incident> inc = {
        {2,5,"Accident"},
        {4,8,"Road Block"},
        {1,3,"Traffic Jam"}
    };

    mergeSort(inc,0,inc.size()-1);

    cout << "\\nIncidents Sorted by Severity:\\n";
    for(auto i: inc)
        cout << i.description << " - " << i.severity << endl;

    return 0;
}
