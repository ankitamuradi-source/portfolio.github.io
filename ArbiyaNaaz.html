1. Grocery Delivery App

Income Model: Delivery charges + store commission.

SDG: Responsible Consumption & Production (SDG 12)

DSA Used: Queue & HashMap

Reason: Queue handles deliveries in order; HashMap provides fast inventory lookup.

#include <iostream>
#include <queue>
#include <unordered_map>
using namespace std;

struct Order { int orderId; string item; };

int main() {
    queue<Order> deliveryQueue;
    unordered_map<string,int> storeInventory;

    storeInventory["Apple"]=50; storeInventory["Milk"]=20;
    deliveryQueue.push({1,"Apple"}); deliveryQueue.push({2,"Milk"});

    while(!deliveryQueue.empty()) {
        Order o = deliveryQueue.front(); deliveryQueue.pop();
        if(storeInventory[o.item]>0){
            cout << "Delivered: " << o.item << " for order " << o.orderId << endl;
            storeInventory[o.item]--;
        } else {
            cout << o.item << " is out of stock!" << endl;
        }
    }
    return 0;
}

2. Online Tutoring Platform

Income Model: Subscription fees + commission per class.

SDG: Quality Education (SDG 4)

DSA Used: Priority Queue & HashMap

Reason: Priority Queue schedules tutors by rating; HashMap manages student records quickly.

#include <iostream>
#include <queue>
#include <unordered_map>
using namespace std;

struct Tutor { string name; int rating; };
struct Compare { bool operator()(Tutor const &a, Tutor const &b){ return a.rating < b.rating; }};

int main() {
    priority_queue<Tutor, vector<Tutor>, Compare> tutorQueue;
    unordered_map<string,string> studentRecords;

    tutorQueue.push({"Alice",95}); tutorQueue.push({"Bob",90});
    studentRecords["John"]="Math"; studentRecords["Emma"]="Physics";

    while(!tutorQueue.empty()){
        Tutor t = tutorQueue.top(); tutorQueue.pop();
        cout << "Assigned Tutor: " << t.name << " with rating " << t.rating << endl;
    }

    for(auto s: studentRecords)
        cout << "Student " << s.first << " learning " << s.second << endl;

    return 0;
}

3. Ride-Sharing App

Income Model: Commission per ride + surge pricing.

SDG: Sustainable Cities & Communities (SDG 11)

DSA Used: Graph & Dijkstra's Algorithm

Reason: Graph represents roads; Dijkstra finds shortest path for efficient rides.

#include <iostream>
#include <vector>
#include <queue>
using namespace std;

struct Edge { int to, weight; };
vector<vector<Edge>> graph;

vector<int> dijkstra(int n, int start){
    vector<int> dist(n, 1e9); dist[start]=0;
    priority_queue<pair<int,int>, vector<pair<int,int>>, greater<pair<int,int>>> pq;
    pq.push({0,start});
    while(!pq.empty()){
        int d = pq.top().first, u = pq.top().second; pq.pop();
        if(d>dist[u]) continue;
        for(auto e: graph[u]){
            if(dist[u]+e.weight<dist[e.to]){
                dist[e.to]=dist[u]+e.weight;
                pq.push({dist[e.to],e.to});
            }
        }
    }
    return dist;
}

int main() {
    int n=5; graph.resize(n);
    graph[0].push_back({1,10}); graph[0].push_back({2,3});
    graph[2].push_back({1,1}); graph[1].push_back({3,2});

    vector<int> shortest = dijkstra(n,0);
    for(int i=0;i<n;i++) cout << "Distance to " << i << ": " << shortest[i] << endl;
}

4. Food Waste Reduction App

Income Model: Subscription fees + ads.

SDG: Responsible Consumption & Production (SDG 12)

DSA Used: HashMap & Heap

Reason: Heap tracks expiring items; HashMap stores food inventory efficiently.

#include <iostream>
#include <queue>
#include <unordered_map>
using namespace std;

struct Food { string name; int expiry; };
struct Compare { bool operator()(Food const &a, Food const &b){ return a.expiry > b.expiry; }};

int main(){
    priority_queue<Food, vector<Food>, Compare> foodHeap;
    unordered_map<string,int> inventory;

    inventory["Bread"]=10; inventory["Milk"]=5;
    foodHeap.push({"Bread",2}); foodHeap.push({"Milk",1});

    while(!foodHeap.empty()){
        Food f = foodHeap.top(); foodHeap.pop();
        if(inventory[f.name]>0){
            cout << "Distribute: " << f.name << " expiring in " << f.expiry << " days" << endl;
            inventory[f.name]--;
        }
    }
}

6. E-Library Platform

Income Model: Membership fees + pay-per-book.

SDG: Quality Education (SDG 4)

DSA Used: Trie & HashMap

Reason: Trie helps in fast book search; HashMap tracks user borrowing history.

#include <iostream>
#include <unordered_map>
using namespace std;

struct TrieNode{
    unordered_map<char,TrieNode*> children;
    bool isEnd=false;
};

void insert(TrieNode* root, string word){
    TrieNode* node=root;
    for(char c: word){
        if(!node->children[c]) node->children[c]=new TrieNode();
        node=node->children[c];
    }
    node->isEnd=true;
}

bool search(TrieNode* root, string word){
    TrieNode* node=root;
    for(char c: word){
        if(!node->children[c]) return false;
        node=node->children[c];
    }
    return node->isEnd;
}

int main(){
    TrieNode* root=new TrieNode();
    insert(root,"Physics"); insert(root,"Math"); insert(root,"Biology");

    cout << "Search Math: " << (search(root,"Math")?"Found":"Not Found") << endl;
}

9. Smart Home Automation

Income Model: Device sales + subscription for premium features.

SDG: Industry, Innovation & Infrastructure (SDG 9)

DSA Used: Graph & DFS

Reason: Graph models devices & connections; DFS triggers automated tasks efficiently.

#include <iostream>
#include <vector>
using namespace std;

vector<vector<int>> devices;
vector<bool> visited;

void dfs(int u){
    visited[u]=true;
    cout << "Activating device " << u << endl;
    for(int v: devices[u]) if(!visited[v]) dfs(v);
}

int main(){
    int n=4; devices.resize(n); visited.resize(n,false);
    devices[0]={1,2}; devices[1]={3};

    dfs(0);
}

struct Compare{ bool operator()(Event a, Event b){ return a.popularity<b.popularity; }};

    }
}
