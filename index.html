// Example: Dijkstra's Algorithm in C++
#include <iostream>
#include <vector>
#include <queue>
using namespace std;
struct Edge { int to, weight; };
void dijkstra(int src, vector<vector<Edge>> &graph) {
    int n = graph.size();
    vector<int> dist(n, 1e9);
    dist[src] = 0;
    priority_queue<pair<int,int>, vector<pair<int,int>>, greater<>> pq;
    pq.push({0, src});
    while(!pq.empty()){
        auto [d,u] = pq.top(); pq.pop();
        if(d > dist[u]) continue;
        for(auto e: graph[u]){
            if(dist[e.to] > dist[u]+e.weight){
                dist[e.to] = dist[u]+e.weight;
                pq.push({dist[e.to], e.to});
            }
        }
    }
}

// Example: Floyd-Warshall Algorithm
#include <iostream>
#include <vector>
using namespace std;
void floydWarshall(vector<vector<int>> &graph){
    int n = graph.size();
    for(int k=0;k<n;k++)
        for(int i=0;i<n;i++)
            for(int j=0;j<n;j++)
                if(graph[i][k]!=1e9 && graph[k][j]!=1e9)
                    graph[i][j] = min(graph[i][j], graph[i][k]+graph[k][j]);
}

// Example: Kruskal's Algorithm for MST
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
struct Edge{int u,v,w;};
int findSet(int u, vector<int>& parent){ return parent[u]==u?u:parent[u]=findSet(parent[u], parent);}
void kruskal(vector<Edge> &edges, int n){
    sort(edges.begin(), edges.end(), [](Edge a, Edge b){return a.w<b.w;});
    vector<int> parent(n);
    for(int i=0;i<n;i++) parent[i]=i;
    for(auto e: edges){
        int u=findSet(e.u,parent), v=findSet(e.v,parent);
        if(u!=v) { parent[u]=v; cout<<e.u<<"-"<<e.v<<" "<<e.w<<endl;}
    }
}

// Example: Trie for Fast Word Search
#include <iostream>
#include <map>
using namespace std;
struct TrieNode{ map<char,TrieNode*> children; bool end=false; };
void insert(TrieNode* root, string s){
    for(char c:s){ if(!root->children[c]) root->children[c]=new TrieNode(); root=root->children[c]; }
    root->end=true;
}
bool search(TrieNode* root, string s){
    for(char c:s){ if(!root->children[c]) return false; root=root->children[c]; }
    return root->end;
}

// Example: Rabin-Karp String Matching
#include <iostream>
#include <string>
using namespace std;
void rabinKarp(string text, string pat){
    int n=text.size(), m=pat.size(), p=31, mod=1e9+7;
    long long pat_hash=0, text_hash=0, p_pow=1;
    for(int i=0;i<m;i++){ pat_hash=(pat_hash+(pat[i]-'a'+1)*p_pow)%mod; p_pow=(p_pow*p)%mod;}
    p_pow=1;
    for(int i=0;i<m;i++){ text_hash=(text_hash+(text[i]-'a'+1)*p_pow)%mod; p_pow=(p_pow*p)%mod;}
    for(int i=0;i<=n-m;i++){
        if(pat_hash==text_hash) cout<<"Pattern at "<<i<<endl;
        if(i<n-m){ text_hash=(text_hash-(text[i]-'a'+1)+mod)%mod; text_hash=(text_hash/p)%mod;
        text_hash=(text_hash+(text[i+m]-'a'+1)*p_pow)%mod;}
    }
}

// Example: DP for Maximum Yield
#include <iostream>
#include <vector>
using namespace std;
int maximizeYield(vector<int> crops, int n){
    vector<int> dp(n+1,0); dp[0]=0; dp[1]=crops[0];
    for(int i=2;i<=n;i++) dp[i]=max(dp[i-1], dp[i-2]+crops[i-1]);
    return dp[n];
}

#include <queue>
using namespace std;
void bfs(int src, vector<vector<int>> &graph){
    int n = graph.size(); vector<bool> vis(n,false); queue<int> q; q.push(src); vis[src]=true;
    while(!q.empty()){ int u=q.front(); q.pop(); for(int v: graph[u]) if(!vis[v]){ vis[v]=true; q.push(v); } }

// Example: Binary Search for Inventory Lookup
#include <iostream>
#include <vector>
using namespace std;
int binarySearch(vector<int>& arr, int key){
    int l=0, r=arr.size()-1;
    while(l<=r){ int m=(l+r)/2; if(arr[m]==key) return m; else if(arr[m]<key) l=m+1; else r=m-1;}
    return -1;
}
