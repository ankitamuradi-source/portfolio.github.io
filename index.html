// Dijkstra: shortest-path template
#include <iostream>
#include <vector>
#include <queue>
using namespace std;
using pii = pair;
vector dijkstra(int n, vector>& g, int src){
  const int INF = 1e9; vector dist(n,INF);
  priority_queue, greater<pii>> pq;
  dist[src]=0; pq.push({0,src});
  while(!pq.empty()){
    auto [d,u]=pq.top(); pq.pop();
    if(d>dist[u]) continue;
    for(auto &e: g[u]){
      int v=e.first,w=e.second;
      if(dist[u]+w < dist[v]){ dist[v]=dist[u]+w; pq.push({dist[v],v}); }
    }
  }
  return dist;
}

// 0/1 Knapsack (DP)
#include <iostream>
#include <vector>
using namespace std;
int knap(int W, vector& wt, vector& val){
  int n=wt.size(); vector dp(W+1,0);
  for(int i=0;i<n;i++)
    for(int w=W; w>=wt[i]; w--) dp[w]=max(dp[w], dp[w-wt[i]]+val[i]);
  return dp[W];
}

// Simple moving average
#include <iostream>
#include <vector>
using namespace std;
double movingAvg(const vector<double>& a, int k){
  int n=a.size(); double sum=0;
  for(int i=0;i<k;i++) sum+=a[i];
  double best=sum/k;
  for(int i=k;i<n;i++){ sum+=a[i]-a[i-k]; best=sum/k; }
  return best;
}

// Simple linear regression (least squares) skeleton
#include <iostream>
#include <vector>
using namespace std;
pair<double,double> linearFit(const vector<double>& x,const vector<double>&y){
  int n=x.size(); double sx=0,sy=0,sxy=0,sxx=0;
  for(int i=0;i<n;i++){ sx+=x[i]; sy+=y[i]; sxy+=x[i]*y[i]; sxx+=x[i]*x[i]; }
  double m = (n*sxy - sx*sy) / (n*sxx - sx*sx);
  double b = (sy - m*sx)/n; return {m,b};
}

// Topological sort (Kahn's algorithm)
#include <iostream>
#include <vector>
#include <queue>
using namespace std;
vector topo(int n, vector>& g){
  vector indeg(n,0), res; for(int u=0;u<n;u++) for(int v:g[u]) indeg[v]++;
  queueq; for(int i=0;i<n;i++) if(indeg[i]==0) q.push(i);
  while(!q.empty()){ int u=q.front(); q.pop(); res.push_back(u);
    for(int v:g[u]) if(--indeg[v]==0) q.push(v);
  } return res;
}

// KMP pattern search
#include <iostream>
#include <vector>
#include <string>
using namespace std;
vector<int> prefix(const string &s){ int n=s.size(); vector<int> pi(n);
  for(int i=1;i<n;i++){ int j=pi[i-1]; while(j>0 && s[i]!=s[j]) j=pi[j-1]; if(s[i]==s[j]) j++; pi[i]=j; } return pi;
}
vector<int> kmp(const string &t,const string &p){
  string s=p+'#'+t; auto pi=prefix(s); vector ans;
  for(int i=0;i<s.size();i++) if(pi[i]==p.size()) ans.push_back(i-2*p.size());
  return ans;
}
